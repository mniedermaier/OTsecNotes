\posterwarning{
Most environments should target \textbf{SL2} as baseline. Critical infrastructure (energy, water, chemical) typically requires \textbf{SL3}. SL4 is reserved for high-value national security targets.
}

\section{\textcolor{accent}{\faIcon{map}}\hspace{0.4em}Implementation Guidance}

\postersuccess{
\textbf{Assessment approach:}
\begin{enumerate}
    \item \textcolor{success}{\faIcon{check}}\hspace{0.2em}\textbf{Map zones and conduits} -- Document OT architecture
    \item \textcolor{success}{\faIcon{check}}\hspace{0.2em}\textbf{Determine target SL} -- Based on risk assessment
    \item \textcolor{success}{\faIcon{check}}\hspace{0.2em}\textbf{Gap analysis} -- Compare current state to target
    \item \textcolor{success}{\faIcon{check}}\hspace{0.2em}\textbf{Prioritize} -- Address highest-risk gaps first
    \item \textcolor{success}{\faIcon{check}}\hspace{0.2em}\textbf{Implement incrementally} -- Don't try SL3 overnight
\end{enumerate}
}

\posterdanger{
\textbf{Common pitfalls:} Applying IT solutions without OT adaptation; targeting SL4 when SL2 is sufficient; ignoring operational impact of security controls; focusing on technology without processes and training.
}
