\section{Introduction}
% ============================================================================

\begin{infobox}
IEC 62443 defines four Security Levels (SL1--SL4) that represent increasing levels of protection against different threat actors. This document maps practical security solutions to each level.
\end{infobox}

\subsection{Security Level Definitions}

\begin{table}[H]
\centering
\small
\begin{tabularx}{\textwidth}{|c|l|X|}
\hline
\textbf{SL} & \textbf{Threat Actor} & \textbf{Description} \\
\hline
\slone & Casual/Accidental & Protection against unintentional errors \\
\sltwo & Intentional, Low Resources & Script kiddies, disgruntled employees \\
\slthree & Intentional, Moderate Resources & Hacktivists, organized crime \\
\slfour & Intentional, High Resources & Nation-states, APT groups \\
\hline
\end{tabularx}
\caption{IEC 62443 Security Levels}
\end{table}

\begin{warningbox}
Most industrial environments should target \textbf{SL2} as a baseline. Critical infrastructure (energy, water, chemical) typically requires \textbf{SL3}. \textbf{SL4} is reserved for high-value national security targets.
\end{warningbox}

IEC 62443-3-3 defines seven Foundational Requirements (FR):

\begin{table}[H]
\centering
\small
\begin{tabularx}{\textwidth}{|c|l|X|}
\hline
\textbf{FR} & \textbf{Name} & \textbf{Focus Area} \\
\hline
FR1 & Identification \& Authentication & User and device identity \\
FR2 & Use Control & Authorization and privileges \\
FR3 & System Integrity & Protection from tampering \\
FR4 & Data Confidentiality & Protection of sensitive data \\
FR5 & Restricted Data Flow & Network segmentation \\
FR6 & Timely Response to Events & Monitoring and incident response \\
FR7 & Resource Availability & Denial of service protection \\
\hline
\end{tabularx}
\caption{IEC 62443 Foundational Requirements}
\end{table}

\section{FR1: Identification \& Authentication}
% ============================================================================

\begin{definitionbox}{FR1 -- Identification \& Authentication}
Controls for identifying and authenticating users, processes, and devices before granting access to system resources.
\end{definitionbox}

\begin{table}[H]
\centering
\small
\begin{tabularx}{\textwidth}{|l|X|}
\hline
\textbf{Level} & \textbf{Security Solutions} \\
\hline
\slone &
\begin{minipage}[t]{\linewidth}
\vspace{2pt}
\begin{itemize}[nosep,leftmargin=*]
    \item Unique user accounts (no shared accounts)
    \item Basic password policy (8+ characters)
    \item Device identification by IP/hostname
\end{itemize}
\vspace{2pt}
\end{minipage} \\
\hline
\sltwo &
\begin{minipage}[t]{\linewidth}
\vspace{2pt}
\begin{itemize}[nosep,leftmargin=*]
    \item Strong password policy (complexity, expiration)
    \item Role-based access control (RBAC)
    \item Central authentication (Active Directory)
    \item Device certificates for critical systems
\end{itemize}
\vspace{2pt}
\end{minipage} \\
\hline
\slthree &
\begin{minipage}[t]{\linewidth}
\vspace{2pt}
\begin{itemize}[nosep,leftmargin=*]
    \item Multi-factor authentication (MFA)
    \item Hardware tokens or smart cards
    \item Certificate-based device authentication
    \item Privileged Access Management (PAM)
\end{itemize}
\vspace{2pt}
\end{minipage} \\
\hline
\slfour &
\begin{minipage}[t]{\linewidth}
\vspace{2pt}
\begin{itemize}[nosep,leftmargin=*]
    \item Biometric authentication
    \item Hardware Security Modules (HSM)
    \item Mutual TLS for all connections
    \item Continuous authentication/verification
\end{itemize}
\vspace{2pt}
\end{minipage} \\
\hline
\end{tabularx}
\caption{FR1 Solutions by Security Level}

\section{Implementation Guidance}
% ============================================================================

\begin{successbox}
Start with an SL assessment to determine your target level, then use this mapping to select appropriate security solutions. Not every control is needed---focus on your specific risks and asset criticality.
\end{successbox}

\subsection{Assessment Approach}

\begin{enumerate}
    \item \textbf{Identify zones and conduits} -- Map your OT architecture
    \item \textbf{Determine target SL} -- Based on risk assessment and threat analysis
    \item \textbf{Gap analysis} -- Compare current state to target SL requirements
    \item \textbf{Prioritize} -- Address highest-risk gaps first
    \item \textbf{Implement incrementally} -- Don't try to reach SL3 overnight
\end{enumerate}

\subsection{Common Pitfalls}

\begin{itemize}
    \item Applying IT solutions without OT adaptation
    \item Targeting SL4 when SL2 is sufficient
    \item Ignoring operational impact of security controls
    \item Focusing on technology without processes and training
\end{itemize}

\section{Summary}
% ============================================================================

\begin{definitionbox}{Key Takeaways}
\begin{itemize}
    \item \textbf{SL1--SL4} -- Increasing protection against sophisticated threats
    \item \textbf{7 Foundational Requirements} -- Comprehensive security coverage
    \item \textbf{SL2 baseline} -- Appropriate for most industrial environments
    \item \textbf{SL3 for critical infrastructure} -- Energy, water, chemical sectors
    \item \textbf{Incremental approach} -- Build security maturity over time
    \item \textbf{Balance security and operations} -- OT availability is paramount
\end{itemize}
\end{definitionbox}
