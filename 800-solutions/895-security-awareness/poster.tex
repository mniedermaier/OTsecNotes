\section{Why OT-Specific Training?}

\posterinfo{
Generic IT training is insufficient for OT:
\begin{itemize}
    \item Safety implications of security incidents
    \item USB critical for air-gapped systems
    \item Physical access threats are common
    \item Diverse workforce (operators, engineers)
    \item 24/7 shift operations
\end{itemize}
}

\section{Key Training Topics}

\posterdanger{
\textbf{USB/Removable Media}
\begin{itemize}
    \item Never use unknown USB devices
    \item Scan all media before use
    \item Beware of ``USB drop'' attacks
    \item Use only approved devices
\end{itemize}
}

\posterwarning{
\textbf{Social Engineering}
\begin{itemize}
    \item Pretexting (false scenarios)
    \item Tailgating into secure areas
    \item Impersonation of vendors/IT
    \item Always verify identities
\end{itemize}
}

\textbf{Physical Security:}
\begin{itemize}
    \item Badge usage and door security
    \item Clean desk policy
    \item Lock workstations
    \item Report suspicious activity
\end{itemize}

\textbf{Passwords:}
\begin{itemize}
    \item Strong, unique passwords
    \item Never share credentials
    \item Understand shared account risks
\end{itemize}

\section{Delivery Methods}

\postersuccess{
\begin{itemize}
    \item \textbf{Classroom} -- Onboarding, complex topics
    \item \textbf{E-Learning} -- Flexible, shift-friendly
    \item \textbf{Hands-on} -- Phishing tests, USB drops
    \item \textbf{Just-in-time} -- Toolbox talks, posters
\end{itemize}
}

\section{Building Security Culture}

\textbf{Leadership:}
\begin{itemize}
    \item Visible management support
    \item Security in regular meetings
    \item Adequate training resources
\end{itemize}

\textbf{Positive Reinforcement:}
\begin{itemize}
    \item Reward reporting concerns
    \item Recognize threat identifiers
    \item No blame for honest mistakes
\end{itemize}

\posterinfo{
\begin{itemize}
    \item \textbf{IEC 62443-2-1} -- Awareness programs
    \item \textbf{NERC CIP-004} -- Personnel training
    \item \textbf{NIST 800-82} -- OT training guidance
    \item \textbf{NIS2} -- Cybersecurity training
\end{itemize}
}

\section{Key Takeaways}

\postersuccess{
\begin{itemize}
    \item OT training must cover USB and physical threats
    \item Tailor content to roles (operators vs engineers)
    \item Use multiple delivery methods for shift workers
    \item Build culture through leadership and reinforcement
    \item Measure with phishing tests and reporting rates
\end{itemize}
}
