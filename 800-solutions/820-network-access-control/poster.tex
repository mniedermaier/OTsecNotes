\section{\textcolor{accent}{\faIcon{info-circle}}\hspace{0.4em}Overview}

Network Access Control (NAC) ensures only \textbf{authorized and compliant devices} can connect to the OT network. It provides visibility into connected assets and enforces security policies before granting access.

\posterinfo{
\textbf{Core capabilities:} \textcolor{accent}{\faIcon{user-lock}}\hspace{0.2em}Authentication (verify identity), \textcolor{accent}{\faIcon{user-lock}}\hspace{0.2em}authorization (determine access level), assessment (check compliance), remediation (quarantine non-compliant), and visibility (inventory all devices).
}

\posterdanger{
\textbf{Why traditional NAC fails in OT:} PLCs/RTUs cannot run agents. Legacy devices don't support 802.1X. Authentication delays may disrupt real-time control. Blocking a device could stop production. Many OT devices use static IPs.
}

\section{\textcolor{accent}{\faIcon{cogs}}\hspace{0.4em}Implementation Phases}

\subsection{\textcolor{accent}{\faIcon{eye}}\hspace{0.3em}Phase 1: Visibility}

\begin{itemize}
    \item \textcolor{accent}{\faIcon{eye}}\hspace{0.2em}Deploy passive monitoring (no enforcement)
    \item \textcolor{accent}{\faIcon{list-alt}}\hspace{0.2em}Build complete asset inventory
    \item \textcolor{accent}{\faIcon{fingerprint}}\hspace{0.2em}Profile all device types and behaviors
    \item \textcolor{accent}{\faIcon{bell}}\hspace{0.2em}Alert only on unknown devices
\end{itemize}

\subsection{\textcolor{accent}{\faIcon{tags}}\hspace{0.3em}Phase 2: Classification}

\begin{itemize}
    \item \textcolor{accent}{\faIcon{tags}}\hspace{0.2em}Categorize devices by type and criticality
    \item \textcolor{accent}{\faIcon{clipboard-list}}\hspace{0.2em}Define policy groups per zone
    \item \textcolor{accent}{\faIcon{barcode}}\hspace{0.2em}Document MAC addresses for MAB enrollment
\end{itemize}

\subsection{\textcolor{accent}{\faIcon{shield-alt}}\hspace{0.3em}Phase 3: Selective Enforcement}

\begin{itemize}
    \item \textcolor{accent}{\faIcon{key}}\hspace{0.2em}Enable 802.1X on IT-like OT devices
    \item \textcolor{accent}{\faIcon{ethernet}}\hspace{0.2em}Implement MAB for industrial devices
    \item \textcolor{accent}{\faIcon{flask}}\hspace{0.2em}Start with low-risk zones first
    \item \textcolor{accent}{\faIcon{heartbeat}}\hspace{0.2em}Monitor continuously for disruption
\end{itemize}

\subsection{\textcolor{accent}{\faIcon{network-wired}}\hspace{0.3em}Phase 4: Full Deployment}

\begin{itemize}
    \item \textcolor{accent}{\faIcon{expand-arrows-alt}}\hspace{0.2em}Extend to all zones (except safety)
    \item \textcolor{accent}{\faIcon{ban}}\hspace{0.2em}Automate quarantine for unknown devices
    \item \textcolor{accent}{\faIcon{project-diagram}}\hspace{0.2em}Integrate with incident response workflows
\end{itemize}

\posterwarning{
\textbf{Start in monitor mode.} Always deploy passive monitoring before enabling enforcement. A phased approach prevents disruption from incomplete device profiling.
}

\section{\textcolor{accent}{\faIcon{fingerprint}}\hspace{0.4em}Device Profiling Techniques}

\begin{itemize}
    \item \textcolor{accent}{\faIcon{network-wired}}\hspace{0.2em}\textbf{DHCP fingerprinting} -- Identify by DHCP options
    \item \textcolor{accent}{\faIcon{barcode}}\hspace{0.2em}\textbf{MAC OUI lookup} -- Manufacturer identification
    \item \textcolor{accent}{\faIcon{chart-line}}\hspace{0.2em}\textbf{Traffic analysis} -- Protocol and behavior patterns
    \item \textcolor{accent}{\faIcon{search}}\hspace{0.2em}\textbf{Active scanning} -- Probe devices (use carefully in OT)
    \item \textcolor{accent}{\faIcon{plug}}\hspace{0.2em}\textbf{Integration} -- Import from asset management tools
\end{itemize}

\postersuccess{
\textbf{Integration points:} SIEM (event correlation), asset management (inventory sync), CMDB (authorized device validation), firewall (dynamic policy updates), vulnerability scanner (trigger on new devices), ticketing (auto-create violation tickets).
}

\postertip{
Use \textbf{agentless, passive} profiling for OT devices. Deploy out-of-band monitoring rather than inline blocking. Use 802.1X only for capable devices and MAC Authentication Bypass for PLCs and RTUs. Different zones need different NAC modes---never block critical control devices. NAC provides essential asset visibility even before enforcement is enabled.
}
