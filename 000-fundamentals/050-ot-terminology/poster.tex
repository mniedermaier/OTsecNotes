\section{\textcolor{accent}{\faIcon{microchip}}\hspace{0.4em}Hardware Components}

\begin{itemize}
    \item \textcolor{accent}{\faIcon{thermometer-half}}\hspace{0.2em}\textbf{Sensor:} Measures physical parameters (temp, pressure, flow, level)
    \item \textcolor{accent}{\faIcon{broadcast-tower}}\hspace{0.2em}\textbf{Transmitter:} Sensor with built-in signal conditioning and communication
    \item \textcolor{accent}{\faIcon{cog}}\hspace{0.2em}\textbf{Actuator:} Performs physical action (valve, motor, relay)
    \item \textcolor{accent}{\faIcon{tachometer-alt}}\hspace{0.2em}\textbf{VFD/VSD:} Variable Frequency/Speed Drive -- controls motor speed
    \item \textcolor{accent}{\faIcon{plug}}\hspace{0.2em}\textbf{I/O Module:} Interface between controller and field devices
    \item \textcolor{accent}{\faIcon{database}}\hspace{0.2em}\textbf{Historian:} Server storing time-series process data
    \item \textcolor{accent}{\faIcon{laptop}}\hspace{0.2em}\textbf{Eng. Workstation:} PC for programming controllers
\end{itemize}
