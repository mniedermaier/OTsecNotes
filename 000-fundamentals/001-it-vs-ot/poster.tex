\section{\textcolor{accent}{\faIcon{book}}\hspace{0.4em}Definitions}

\posterdefinition{Information Technology (IT)}{
Hardware, software, and networks used to process, store, and transmit \textbf{data and information}. Examples: servers, workstations, enterprise apps (ERP, CRM), databases, corporate networks.
}

\posterdefinition{Operational Technology (OT)}{
Hardware, software, and networks used to monitor and control \textbf{physical processes and infrastructure}. Examples: PLCs, SCADA, HMIs, DCS, industrial robots, building automation.
}

\posterwarning{
OT systems often run for 15--25 years. Windows XP and Windows 7 are still common in critical infrastructure because control system vendors only certify those platforms.
}

\posterdanger{
Many OT protocols were designed decades ago for isolated networks. They lack authentication, encryption, and integrity checking. A single packet can command a PLC to perform dangerous actions.
}

\subsection{Convergence Challenges}

\begin{itemize}
    \item Different priorities and cultures between teams
    \item Legacy systems without security features
    \item Expanded attack surface through connectivity
    \item Skill gaps in both IT and OT domains
\end{itemize}

\postertip{
\textbf{The Air Gap Myth:} Most OT environments have some IT connection: historian replication, vendor remote access, USB drives, dual-homed workstations, or cloud monitoring. True air gaps are rare.
}

\section{\textcolor{accent}{\faIcon{shield-alt}}\hspace{0.4em}Security Implications}

\begin{enumerate}
    \item \textcolor{danger}{\faIcon{hard-hat}}\hspace{0.2em}\textbf{Safety first} -- security must not compromise safety
    \item \textcolor{warning}{\faIcon{clock}}\hspace{0.2em}\textbf{No downtime} -- systems run 24/7/365
    \item \textcolor{info}{\faIcon{flask}}\hspace{0.2em}\textbf{No testing} -- cannot pentest production
    \item \textcolor{accent}{\faIcon{lock}}\hspace{0.2em}\textbf{Vendor lock} -- changes void warranties
    \item \textcolor{accent}{\faIcon{history}}\hspace{0.2em}\textbf{Long lifecycles} -- secure for decades
    \item \textcolor{danger}{\faIcon{bolt}}\hspace{0.2em}\textbf{Physical harm} -- cyber causes real damage
\end{enumerate}

\posterdanger{
\textbf{Applying IT practices directly to OT is dangerous:}
\begin{itemize}
    \item Automatic updates may crash control systems
    \item Antivirus scans cause CPU spikes, missed deadlines
    \item Active vulnerability scanning disrupts PLCs
    \item Network changes break process dependencies
\end{itemize}
}

\section{\textcolor{accent}{\faIcon{handshake}}\hspace{0.4em}Bridging the Gap}

\postersuccess{
\textbf{Organizational:} Joint IT/OT governance, cross-training, shared RACI, unified (adapted) policies.

\textbf{Technical:} Network segmentation (Purdue Model), industrial DMZ, secure remote access (jump servers, MFA), OT-specific monitoring, passive asset inventory, compensating controls.
}

\posterdefinition{Key Standards}{
\textbf{IEC 62443} -- Industrial cybersecurity framework (zones, conduits, security levels).
\textbf{NIST SP 800-82} -- Guide to OT security.
\textbf{NERC CIP} -- North American electric grid standards.
}
