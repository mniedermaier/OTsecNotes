\section{\textcolor{accent}{\faIcon{wifi}}\hspace{0.4em}Overview}

Wireless technologies are increasingly deployed in OT for mobility, flexibility, and cost reduction. However, they introduce \textbf{unique security challenges}: wireless extends the attack surface beyond the physical perimeter. An attacker only needs to be within radio range.

WPA2-Personal & AES-CCMP & \textcolor{warning}{\faIcon{exclamation}} Low-risk \\
WPA2-Enterprise & AES + 802.1X & \textcolor{success}{\faIcon{check}} Recommended \\

\end{tabular}
\end{center}

\postersuccess{
\faIcon{check-circle}\hspace{0.2em}\textbf{Minimum for OT WiFi:} WPA2-Enterprise or WPA3, 802.1X with RADIUS, dedicated SSIDs for OT, certificate-based auth where possible, wireless IDS/IPS.
}

\section{\textcolor{accent}{\faIcon{broadcast-tower}}\hspace{0.4em}5G and Private LTE}

\begin{itemize}
    \item \faIcon{lock}\hspace{0.2em}\textbf{Private networks:} Dedicated spectrum, full control
    \item \faIcon{tachometer-alt}\hspace{0.2em}\textbf{Low latency:} Suitable for real-time control
    \item \faIcon{check-double}\hspace{0.2em}\textbf{High reliability:} Carrier-grade availability
    \item \faIcon{map-marked-alt}\hspace{0.2em}\textbf{Wide coverage:} Large facilities, remote sites
    \item \faIcon{sim-card}\hspace{0.2em}\textbf{SIM-based auth:} Built-in device authentication
\end{itemize}

\posterwarning{
\faIcon{exclamation-triangle}\hspace{0.2em}For 5G: use \textbf{private networks} for critical systems. Implement network slicing. Monitor for IMSI catchers. Encrypt at application layer.
}

\section{\textcolor{accent}{\faIcon{ban}}\hspace{0.4em}What NOT to Put on Wireless}

\posterdanger{
\faIcon{skull-crossbones}\hspace{0.2em}\textbf{Avoid wireless for:} Safety Instrumented Systems (SIS), emergency shutdown commands, primary control loops, high-speed motion control, any latency/jitter-critical application.
}

\section{\textcolor{accent}{\faIcon{shield-alt}}\hspace{0.4em}Implementation Guidelines}

\begin{enumerate}
    \item \faIcon{project-diagram}\hspace{0.2em}\textbf{Segment} -- Separate SSID/VLAN for OT wireless
    \item \faIcon{shield-alt}\hspace{0.2em}\textbf{Firewall} -- All wireless through OT firewall
    \item \faIcon{fingerprint}\hspace{0.2em}\textbf{Authenticate} -- 802.1X, certs, or strong PSK
    \item \faIcon{lock}\hspace{0.2em}\textbf{Encrypt} -- WPA2-Enterprise min, WPA3 preferred
    \item \faIcon{eye}\hspace{0.2em}\textbf{Monitor} -- Wireless IDS, rogue AP detection
    \item \faIcon{ban}\hspace{0.2em}\textbf{Limit} -- Restrict to non-safety applications
    \item \faIcon{map-marker-alt}\hspace{0.2em}\textbf{Position APs} to minimize signal outside
    \item \faIcon{search}\hspace{0.2em}\textbf{Survey regularly} to detect rogue devices
\end{enumerate}

\subsection{\textcolor{accent}{\faIcon{eye}}\hspace{0.3em}Monitoring and Detection}

\begin{itemize}
    \item \faIcon{shield-alt}\hspace{0.2em}Deploy Wireless Intrusion Detection (WIDS)
    \item \faIcon{map}\hspace{0.2em}Conduct regular wireless site surveys
    \item \faIcon{bell}\hspace{0.2em}Alert on deauthentication floods
    \item \faIcon{chart-line}\hspace{0.2em}Track RF spectrum for jamming attempts
    \item \faIcon{list-alt}\hspace{0.2em}Maintain inventory of authorized wireless devices
\end{itemize}

\postertip{
\faIcon{lightbulb}\hspace{0.2em}Common OT wireless use cases: mobile HMI/tablets, wireless sensor networks, remote asset monitoring (IIoT), temporary maintenance connections. Always treat wireless as an \textbf{untrusted network segment}.
}
