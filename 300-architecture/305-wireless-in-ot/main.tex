\section{Introduction}
% ============================================================================

\begin{infobox}
Wireless technologies are increasingly deployed in OT environments for mobility, flexibility, and cost reduction. However, they introduce unique security challenges that differ significantly from wired networks.
\end{infobox}

Common wireless use cases in OT:
\begin{itemize}
    \item Mobile HMI and tablets for operators
    \item Wireless sensor networks
    \item Remote asset monitoring (IoT/IIoT)
    \item Temporary connections during maintenance
    \item Connectivity in hazardous or hard-to-wire areas
\end{itemize}

\begin{warningbox}
Wireless extends the attack surface beyond the physical perimeter. An attacker doesn't need physical access---they only need to be within radio range.
\end{warningbox}

\subsection{OT-Specific Concerns}

\begin{dangerbox}
\textbf{Critical OT wireless risks:}
\begin{itemize}
    \item Safety system commands over wireless
    \item Unencrypted legacy industrial protocols
    \item Interference with control system timing
    \item Extended attack surface beyond fence line
    \item Difficulty detecting rogue devices
\end{itemize}
\end{dangerbox}

\subsection{WiFi Best Practices}

\begin{successbox}
\textbf{Minimum requirements for OT WiFi:}
\begin{itemize}
    \item WPA2-Enterprise or WPA3
    \item 802.1X authentication with RADIUS
    \item Dedicated SSIDs for OT (separate from corporate)
    \item Certificate-based authentication where possible
    \item Wireless IDS/IPS deployment
\end{itemize}
\end{successbox}

\begin{itemize}
    \item Disable SSID broadcast (limited value but reduces casual discovery)
    \item Use strong, unique passwords (if PSK required)
    \item Enable client isolation where appropriate
    \item Implement MAC filtering as additional layer
    \item Regular wireless surveys to detect rogues
    \item Position APs to minimize signal leakage outside facility
\end{itemize}

\section{5G and Private LTE}
% ============================================================================

\subsection{Benefits for OT}

\begin{itemize}
    \item \textbf{Private networks} -- Dedicated spectrum, full control
    \item \textbf{Low latency} -- Suitable for real-time control
    \item \textbf{High reliability} -- Carrier-grade availability
    \item \textbf{Wide coverage} -- Large facilities, remote sites
    \item \textbf{Built-in security} -- SIM-based authentication
\end{itemize}

\subsection{5G Security Considerations}

\begin{itemize}
    \item Use private 5G networks, not public carriers for critical systems
    \item Implement network slicing for traffic isolation
    \item Secure the 5G core infrastructure
    \item Monitor for IMSI catchers and fake base stations
    \item Encrypt application layer (don't rely only on 5G encryption)
\end{itemize}

\section{LPWAN (LoRaWAN, NB-IoT)}
% ============================================================================

\subsection{Low-Power Wide-Area Networks}

Common for remote sensors and metering:
\begin{itemize}
    \item \textbf{LoRaWAN} -- Long range, low power, open standard
    \item \textbf{NB-IoT} -- Cellular-based, carrier managed
    \item \textbf{Sigfox} -- Ultra-narrow band, limited messages
\end{itemize}

\subsection{LPWAN Security}

\begin{warningbox}
Many LPWAN deployments use default keys or weak security. Always:
\begin{itemize}
    \item Change default application and network keys
    \item Use unique keys per device
    \item Enable encryption (LoRaWAN AES-128)
    \item Implement secure join procedures (OTAA over ABP)
\end{itemize}
\end{warningbox}

\section{Implementation Guidelines}
% ============================================================================

\subsection{OT Wireless Architecture}

\begin{enumerate}
    \item \textbf{Segment} -- Separate SSID/VLAN for OT wireless
    \item \textbf{Firewall} -- All wireless traffic through OT firewall
    \item \textbf{Authenticate} -- 802.1X, certificates, or strong PSK
    \item \textbf{Encrypt} -- WPA2-Enterprise minimum, WPA3 preferred
    \item \textbf{Monitor} -- Wireless IDS, rogue AP detection
    \item \textbf{Limit} -- Restrict wireless to non-safety applications
\end{enumerate}

\subsection{What NOT to Put on Wireless}

\begin{dangerbox}
\textbf{Avoid wireless for:}
\begin{itemize}
    \item Safety Instrumented Systems (SIS)
    \item Emergency shutdown commands
    \item Primary control loops
    \item High-speed motion control
    \item Any application where latency/jitter is critical
\end{itemize}
\end{dangerbox}

% ============================================================================
\section{Monitoring and Detection}
% ============================================================================

\begin{itemize}
    \item Deploy Wireless Intrusion Detection Systems (WIDS)
    \item Conduct regular wireless site surveys
    \item Monitor for unauthorized SSIDs and clients
    \item Alert on deauthentication floods
    \item Track RF spectrum for jamming attempts
    \item Maintain inventory of authorized wireless devices
\end{itemize}

% ============================================================================
\section{Summary}
% ============================================================================

\begin{definitionbox}{Key Takeaways}
\begin{itemize}
    \item \textbf{Extended attack surface} -- Wireless reaches beyond physical perimeter
    \item \textbf{WPA2-Enterprise minimum} -- No WEP, no WPA-Personal in OT
    \item \textbf{Segment OT wireless} -- Separate from corporate networks
    \item \textbf{Monitor continuously} -- Rogue AP and intrusion detection
    \item \textbf{Avoid for safety} -- Don't use wireless for safety-critical functions
    \item \textbf{Industrial protocols} -- WirelessHART, ISA100.11a for process control
\end{itemize}
\end{definitionbox}
