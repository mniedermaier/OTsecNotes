\section{\textcolor{accent}{\faIcon{bolt}}\hspace{0.4em}Overview}

IEC 61850 is the international standard for communication in \textbf{electrical substations}. Enables interoperability between protective relays, circuit breakers, transformers, and control systems. Targeted by the \textbf{Industroyer/CrashOverride} malware.

\posterinfo{
\faIcon{clock}\hspace{0.2em}\textbf{MMS} handles client/server communication. \textbf{GOOSE} provides sub-4ms multicast for protection events. \textbf{SV} streams real-time measurements. \textbf{PTP} (IEEE 1588) provides precision time sync.
}

\section{\textcolor{accent}{\faIcon{exclamation-triangle}}\hspace{0.4em}Security Vulnerabilities}

\posterdanger{
Base IEC 61850 has \textbf{no security mechanisms}. GOOSE and SV are particularly vulnerable -- they operate at Layer 2, must be processed within milliseconds, and accept messages from any source.
}

\subsection{\textcolor{accent}{\faIcon{crosshairs}}\hspace{0.3em}Attack Vectors}

\begin{enumerate}
    \item \textcolor{danger}{\faIcon{bolt}}\hspace{0.2em}\textbf{GOOSE injection:} Fake trip commands \riskcritical
    \item \textcolor{danger}{\faIcon{ban}}\hspace{0.2em}\textbf{GOOSE blocking:} Prevent protection \riskcritical
    \item \textcolor{warning}{\faIcon{chart-line}}\hspace{0.2em}\textbf{SV manipulation:} False measurements \riskhigh
    \item \textcolor{warning}{\faIcon{cog}}\hspace{0.2em}\textbf{MMS exploitation:} Unauthorized config \riskhigh
    \item \textcolor{accent}{\faIcon{clock}}\hspace{0.2em}\textbf{Time sync attacks:} Disrupt PTP \riskmedium
    \item \textcolor{warning}{\faIcon{redo}}\hspace{0.2em}\textbf{Replay attacks:} Capture/replay GOOSE \riskhigh
\end{enumerate}

\posterwarning{
\faIcon{skull-crossbones}\hspace{0.2em}\textbf{Industroyer/CrashOverride} (2016 Ukraine): targeted IEC 61850, opened circuit breakers causing outages. Demonstrates \textbf{real-world attack capability}.
}

\section{\textcolor{accent}{\faIcon{lock}}\hspace{0.4em}IEC 62351 Security}

\postersuccess{
\textbf{IEC 62351} provides security extensions:
\begin{itemize}
    \item \faIcon{lock}\hspace{0.2em}\textbf{Part 3:} TLS for TCP-based protocols (MMS)
    \item \faIcon{signature}\hspace{0.2em}\textbf{Part 6:} Digital signatures for GOOSE and SV
    \item \faIcon{user-shield}\hspace{0.2em}\textbf{Part 8:} Role-based access control
    \item \faIcon{key}\hspace{0.2em}\textbf{Part 9:} Key management
\end{itemize}
}

\section{\textcolor{accent}{\faIcon{shield-alt}}\hspace{0.4em}Security Mitigations}

\begin{itemize}
    \item \textcolor{success}{\faIcon{check}}\hspace{0.2em}\textbf{Physical isolation} -- Separate process/station bus
    \item \faIcon{project-diagram}\hspace{0.2em}\textbf{VLAN segmentation} -- Isolate GOOSE/SV
    \item \faIcon{ethernet}\hspace{0.2em}\textbf{Managed switches} -- Port security
    \item \faIcon{fingerprint}\hspace{0.2em}\textbf{MAC filtering} -- Restrict L2 sources
    \item \faIcon{id-card}\hspace{0.2em}\textbf{802.1X} -- Port-based authentication
    \item \faIcon{door-closed}\hspace{0.2em}\textbf{Physical access control} to network equipment
    \item \faIcon{lock}\hspace{0.2em}\textbf{No remote access} -- or strictly controlled VPN
\end{itemize}
