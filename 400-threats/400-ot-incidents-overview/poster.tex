\section{\textcolor{accent}{\faIcon{info-circle}}\hspace{0.4em}Overview}

Understanding past OT/ICS incidents helps identify vulnerabilities and implement effective defenses. Real-world examples demonstrate the need for OT security investment.

\posterinfo{
\textbf{Why study incidents?} Learn from mistakes, identify recurring patterns, justify security investments, and improve detection by understanding attacker techniques.
}

\section{\textcolor{accent}{\faIcon{crosshairs}}\hspace{0.4em}Attack Categories}

\subsection{\textcolor{accent}{\faIcon{flag}}\hspace{0.3em}Nation-State Attacks}

\begin{itemize}
    \item \textcolor{danger}{\faIcon{skull-crossbones}}\hspace{0.2em}Highly sophisticated and well-resourced
    \item \textcolor{danger}{\faIcon{flag}}\hspace{0.2em}Target critical infrastructure
    \item \textcolor{danger}{\faIcon{chess}}\hspace{0.2em}Focus on espionage, sabotage, or pre-positioning
    \item \textcolor{danger}{\faIcon{fire}}\hspace{0.2em}\textit{Examples:} Stuxnet, Ukraine Grid, TRITON
\end{itemize}

\subsection{\textcolor{accent}{\faIcon{lock}}\hspace{0.3em}Ransomware and Criminal}

\begin{itemize}
    \item \textcolor{warning}{\faIcon{dollar-sign}}\hspace{0.2em}Financially motivated, often opportunistic
    \item \textcolor{warning}{\faIcon{desktop}}\hspace{0.2em}IT systems compromised, OT affected indirectly
    \item \textcolor{warning}{\faIcon{power-off}}\hspace{0.2em}OT shutdown as precaution even if not directly hit
    \item \textcolor{warning}{\faIcon{fire}}\hspace{0.2em}\textit{Examples:} Colonial Pipeline, JBS, Norsk Hydro
\end{itemize}

\subsection{\textcolor{accent}{\faIcon{user-secret}}\hspace{0.3em}Insider Threats}

\begin{itemize}
    \item \textcolor{info}{\faIcon{user-lock}}\hspace{0.2em}Authorized access misused (malicious or unintentional)
    \item \textcolor{info}{\faIcon{eye-slash}}\hspace{0.2em}Difficult to detect with perimeter security alone
    \item \textcolor{info}{\faIcon{database}}\hspace{0.2em}Significant damage due to deep system knowledge
    \item \textcolor{info}{\faIcon{water}}\hspace{0.2em}\textit{Example:} Maroochy Shire sewage spill (2000)
\end{itemize}

\subsection{\textcolor{accent}{\faIcon{sign-in-alt}}\hspace{0.3em}Initial Access Methods}

\begin{enumerate}
    \item \textcolor{danger}{\faIcon{envelope}}\hspace{0.2em}\textbf{Spear Phishing} -- Targeted emails to OT-connected staff
    \item \textcolor{danger}{\faIcon{key}}\hspace{0.2em}\textbf{Remote Access} -- Compromised VPNs, RDP, vendor links
    \item \textcolor{danger}{\faIcon{truck}}\hspace{0.2em}\textbf{Supply Chain} -- Compromised software updates/tools
    \item \textcolor{danger}{\faIcon{usb}}\hspace{0.2em}\textbf{Removable Media} -- USB drives crossing air gaps
    \item \textcolor{danger}{\faIcon{globe}}\hspace{0.2em}\textbf{Internet-Exposed} -- Misconfigured devices online
\end{enumerate}

\subsection{\textcolor{accent}{\faIcon{exchange-alt}}\hspace{0.3em}Lateral Movement to OT}

\begin{enumerate}
    \item \textcolor{warning}{\faIcon{laptop}}\hspace{0.2em}\textbf{Dual-Homed Systems} -- Eng. workstations on both nets
    \item \textcolor{warning}{\faIcon{server}}\hspace{0.2em}\textbf{Historian Servers} -- Data replication IT$\leftrightarrow$OT
    \item \textcolor{warning}{\faIcon{key}}\hspace{0.2em}\textbf{Shared Credentials} -- Same passwords IT and OT
    \item \textcolor{warning}{\faIcon{project-diagram}}\hspace{0.2em}\textbf{Flat Networks} -- No segmentation, direct access
\end{enumerate}

\section{\textcolor{accent}{\faIcon{lightbulb}}\hspace{0.4em}Key Lessons Learned}

\postersuccess{
\textbf{Recurring themes across major incidents:}
\begin{enumerate}
    \item \textcolor{info}{\faIcon{project-diagram}}\hspace{0.2em}Network segmentation failures enabled lateral movement
    \item \textcolor{info}{\faIcon{eye-slash}}\hspace{0.2em}Lack of visibility into OT network traffic
    \item \textcolor{danger}{\faIcon{key}}\hspace{0.2em}Weak authentication and access controls
    \item \textcolor{info}{\faIcon{band-aid}}\hspace{0.2em}Delayed patching of known vulnerabilities
\end{enumerate}
}

\section{\textcolor{accent}{\faIcon{shield-alt}}\hspace{0.4em}Defense Priorities}

\begin{enumerate}
    \item \textcolor{success}{\faIcon{shield-alt}}\hspace{0.2em}\textbf{Segment} -- Proper IT/OT/DMZ boundaries with firewalls
    \item \textcolor{success}{\faIcon{fingerprint}}\hspace{0.2em}\textbf{Authenticate} -- MFA for all remote and privileged access
    \item \textcolor{success}{\faIcon{eye}}\hspace{0.2em}\textbf{Monitor} -- OT network traffic analysis and anomaly detection
    \item \textcolor{success}{\faIcon{tools}}\hspace{0.2em}\textbf{Patch} -- Risk-based patching during maintenance windows
    \item \textcolor{success}{\faIcon{clipboard-list}}\hspace{0.2em}\textbf{Respond} -- OT-specific incident response plans and drills
\end{enumerate}

\posterdanger{
\textbf{Critical:} Most attacks exploited \textbf{known vulnerabilities} and \textbf{basic security gaps} -- not sophisticated zero-days. Weak passwords, missing MFA, and poor segmentation remain top enablers.
}

\postertip{
Study past incidents to identify patterns applicable to your environment -- every major OT attack reinforced the same fundamentals: \textbf{segment, monitor, authenticate, patch, and respond}.
}
