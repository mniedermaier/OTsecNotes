\section{OT-Aware Ransomware}

\posterwarning{
\textbf{EKANS/SNAKE} -- First ICS-aware ransomware
\begin{itemize}
    \item Kills industrial processes before encrypting
    \item Targets: GE Proficy, Honeywell HMI, Fanuc
\end{itemize}
}

\section{Prevention}

\postersuccess{
\textbf{Network:}
\begin{itemize}
    \item IT/OT segmentation with DMZ
    \item Disable/secure RDP
    \item Email filtering + sandboxing
\end{itemize}
\textbf{Endpoint:}
\begin{itemize}
    \item Application whitelisting
    \item EDR where compatible
    \item Regular patching
\end{itemize}
\textbf{Access:}
\begin{itemize}
    \item MFA on all remote access
    \item Privileged access management
    \item Least privilege principle
\end{itemize}
}

\section{Incident Response}

\posterinfo{
\begin{enumerate}
    \item \textbf{Detect} -- Identify scope, preserve logs
    \item \textbf{Contain} -- Isolate systems, protect OT
    \item \textbf{Assess} -- Evaluate damage, legal review
    \item \textbf{Recover} -- Restore from backups
    \item \textbf{Learn} -- Root cause, improve defenses
\end{enumerate}
}

\section{To Pay or Not?}

\posterwarning{
\textbf{Against:} No guarantee, funds criminals, legal risk\\
\textbf{For:} Safety-critical systems, no backup option\\
\textbf{Decision:} Involve legal, business, and technical teams
}

\section{Detection Signs}

\begin{itemize}
    \item Unusual network scanning activity
    \item Mass file modifications
    \item Disabled security tools
    \item Unauthorized lateral movement
    \item Unexpected process termination
\end{itemize}

\section{Reporting Requirements}

\begin{itemize}
    \item \textbf{CIRCIA (US):} 72-hour reporting
    \item \textbf{NIS2 (EU):} 24h warning, 72h report
    \item \textbf{TSA Directives:} Pipeline specific
\end{itemize}

\section{Key Takeaways}

\postersuccess{
\begin{itemize}
    \item OT ransomware causes physical consequences
    \item Most attacks start in IT, spread to OT
    \item Focus: segmentation, MFA, offline backups
    \item Include PLC/HMI configs in backups
    \item Plan incident response before attack
\end{itemize}
}
